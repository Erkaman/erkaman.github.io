In this section, we give a tutorial that demonstrates how easy it is to use the toolkit. First we give the source code of another simple demo:

__global__ void maskKernel(int *g_in, int* g_backwardMask, int* g_forwardMask, int n) {

    //__shared__ int

    for (int i : hemi::grid_stride_range(0, n)) {
        if (i == 0)
            g_backwardMask[i] = 1;
        else {
            g_backwardMask[i] = (g_in[i] != g_in[i - 1]);
        }

        if (i == (n - 1))
            g_forwardMask[i] = 1;
        else {
            g_forwardMask[i] = (g_in[i] != g_in[i + 1]);
        }
    }
}
